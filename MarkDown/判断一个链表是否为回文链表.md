    public ListNode getMiddle(ListNode head){    //返回中间结点
        if(head == null){
            return null;
        }
        ListNode cur = head;
        int steps = getSize(head)/2;
        for(int i =0;i<steps;i++){
            cur = cur.next;
        }
        return cur;
    }
    public ListNode reverseList(ListNode head){
        if(head == null){

            cur = next;
