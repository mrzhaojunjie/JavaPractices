//找到链表中间结点

public static int sizeOFList(Node head){
        Node cur = head;
        int size = 0;
        while(cur!=null){
            cur = cur.next;
            size++;
        }
        return size;
    }

public static Node middleNode(Node head){
        Node cur = head;  
        int length = sizeOFList(head);
        int steps = length/2;
        if(head==null){
            return null;
        }else{
            for(int i = 0;i<steps;i++){  //遍历链表steps次使cur指向链表中间结点
                cur = cur.next;
            }
            return cur;
            }

    }
